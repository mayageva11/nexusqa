// failure-triage.ts 
import { Anthropic } from '@anthropic-ai/sdk';

export async function analyzeFailure(testName: string, errorMsg: string) {
  const anthropic = new Anthropic({ apiKey: process.env.CLAUDE_API_KEY });
  const response = await anthropic.messages.create({
    model: "claude-3-5-sonnet-20241022",
    max_tokens: 150,
    system: "You are a QA automation failure classifier.",
    messages: [{ role: "user", content: `Test: ${testName}\nError: ${errorMsg}` }]
  });
  return JSON.parse(response.content[0].text);
}

How Claude Powers the Tests

The twist that makes this portfolio unique. Claude isn't just a tool I used to write the code — it's an active part of the test infrastructure at runtime.

INTELLIGENT TEST DATA
Claude generates users with real edge cases: unicode names, long emails, special characters, multi-locale inputs. Better than any faker library.
const users = await generateTestUsers(15); // returns users with edge cases: // { name: "Ñoño García-López", // email: "user+tag@sub.domain.io" }
FAILURE ANALYSIS
When a test fails in CI, Claude reads the error and screenshot to classify it: real bug, flaky test, or environment issue. Saved in the Allure report.
const analysis = await analyzeFailure( testName, errorMsg ); // { rootCause: "race condition", // severity: "flaky", // isFlaky: true }
SCENARIO GENERATION
Feed Claude a user story in plain English. It returns a structured test scenario with steps, assertions, and edge cases ready to implement.
const scenario = await generateTestScenario( "user can filter by role" ); // { steps: [...], assertions: [...], // edgeCases: ["empty role", ...] }
